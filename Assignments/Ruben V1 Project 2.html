1.2 Clean Data

Codes are used to indicate observations in which the respondent refused to answer etc. We transform these coded values to missing observations, meaning they cannot interfere with the data accuracy. This is done at the start of the coding process to remove repetitive steps. For Project 2, include 44 and 77.

codes <- c(-9, -8, -5, -3, -2,
           9999, 8888, 5555, 3333, 2222,
           99, 88, 55, 33, 22, 44, 77)

hhderived <- hhderived |>
  mutate(across(
    where(is.numeric),
    ~ replace(.x, .x %in% codes, NA_real_)
  ))

adult <- adult |>

2. Build province and income inputs

What should be done to observations “other” contained in province data? Probably not in SA? So just say that they stay outside of SA? For the income data. If all the income components are missing, then the total remains missing. If at least 1 of the income components is 0 (with the rest being missing), the total is 0.

province_data <- hhderived |>
  select(w2_hhid, w2_prov2011) |>
  mutate(province = as_factor(w2_prov2011, levels = "labels")
  ) |>
  select(w2_hhid, province)

income_components <- indderived |>
  select(w2_fwag:w2_remt) |>
  names()

income_data <- indderived |>
  select(w2_hhid, pid, all_of(income_components)) |>
  mutate(
    income_nonmissing = rowSums(!is.na(across(all_of(income_components)))),
    income_rands = rowSums(across(all_of(income_components)), na.rm = TRUE),
    income_rands = if_else(income_nonmissing == 0, NA_real_, income_rands)
  ) |>
  select(w2_hhid, pid, income_rands)

10. Figure 1: Medical aid coverage

build_age_group <- function(x) {
  case_when(
    x >= 15 & x <= 24 ~ "15-24",
    x >= 25 & x <= 34 ~ "25-34",
    x >= 35 & x <= 44 ~ "35-44",
    x >= 45 & x <= 54 ~ "45-54",
    x >= 55 & x <= 64 ~ "55-64",
    x >= 65 ~ "65+",
    TRUE ~ NA_character_
  )
}

med_summary <- adult_data_development |>
  mutate(
    age_group_plot = build_age_group(age_years),
    age_group_plot = factor(age_group_plot, levels = c("15-24", "25-34", "35-44", "45-54", "55-64", "65+")),
    medical_aid_yes = case_when(
      as.character(medical_aid) == "Yes" ~ 1,
      as.character(medical_aid) == "No" ~ 0,
      TRUE ~ NA_real_
    )
  ) |>
  filter(
    !is.na(age_group_plot),
    !is.na(population_group),
    !is.na(gender),
    !is.na(medical_aid_yes)
  ) |>
  group_by(age_group_plot, population_group, gender) |>
  summarise(
    medical_aid_coverage = mean(medical_aid_yes) * 100,
    n = n(),
    .groups = "drop"
  )

fig1_medical_aid <- med_summary |>
  ggplot(aes(x = age_group_plot, y = medical_aid_coverage, fill = population_group)) +

7. Concentration curve for self-assessed health status

heads <- adult_data_development |>
  filter(head_dummy ==1,
         !is.na(monthly_income_rands),
         !is.na(self_health))


heads <- heads |>
  mutate(self_health_num = case_when(
    self_health == "Excellent" ~ 1,
    self_health == "Very good" ~ 2,
    self_health == "Good" ~ 3,
    self_health == "Fair" ~ 4,
    self_health == "Poor" ~ 5
  ))

heads <- heads |>
  arrange(monthly_income_rands) |>
  mutate(
    rank = row_number(),
    cum_pop = rank / n()
  )

heads <- heads |>
  mutate(
    health_share = self_health_num / sum(self_health_num, na.rm = TRUE),
    cum_health = cumsum(health_share)
  )

ggplot(heads, aes(x = cum_pop, y = cum_health)) +
  geom_line(color = "orange", linewidth = 1) +
  geom_abline(intercept = 0, slope = 1, linetype = "dashed") +
  labs(
    x = "Cumulative Share of Population (Ranked by Income)",
    y = "Cumulative Share of Health (Worse Health)",
    title = "Concentration Curve for Self-Assessed Health"
  ) +
  theme_minimal()
