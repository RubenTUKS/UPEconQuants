nids_missing_num <- c(-9, -8, -5, -3, -2)
nids_missing_year <- c(9999, 8888, 5555, 3333, 2222)
nids_missing_month_day <- c(99, 88, 55, 33, 22)

province_data <- hhderived |>
  select(w2_hhid, w2_prov2011) |>
  mutate(
    w2_prov2011 = replace(w2_prov2011, w2_prov2011 %in% nids_missing_num, NA_real_),
    province = as_factor(w2_prov2011, levels = "labels")
  ) |>
  select(w2_hhid, province)

income_components <- c(
  "w2_fwag", "w2_cwag", "w2_swag", "w2_cheq", "w2_prof", "w2_extr",
  "w2_bonu", "w2_othe", "w2_help", "w2_plot", "w2_opro", "w2_spen",
  "w2_ppen", "w2_uif", "w2_comp", "w2_dis", "w2_chld", "w2_fost",
  "w2_cdep", "w2_indi", "w2_inhe", "w2_rnt", "w2_retr", "w2_brid",
  "w2_gift", "w2_loan", "w2_sale", "w2_remt"
)

income_data <- indderived |>
  select(w2_hhid, pid, all_of(income_components)) |>
  mutate(
    across(
      all_of(income_components),
      ~ case_when(
        .x %in% nids_missing_num ~ NA_real_,
        TRUE ~ as.numeric(.x)
      )
    )
  ) |>
  mutate(
    income_nonmissing = rowSums(!is.na(across(all_of(income_components)))),
    income_rands = rowSums(across(all_of(income_components)), na.rm = TRUE),
    income_rands = if_else(income_nonmissing == 0, NA_real_, income_rands)
  ) |>
  select(w2_hhid, pid, income_rands)

  select(all_of(symptom_vars)) |>

adl_binary <- adult_data_development |>
  select(all_of(adl_vars)) |>
  mutate(
    across(
      everything(),
      ~ case_when(
        .x %in% c(2, 3, 4) ~ 1L,

        TRUE ~ NA_integer_
      )
    )
  )

  mutate(
    symptom_nonmissing = rowSums(!is.na(symptom_binary)),
    symptom_yes_count = rowSums(symptom_binary, na.rm = TRUE),

      symptom_nonmissing == 0 ~ NA_integer_,
      symptom_yes_count > 0 ~ 1L,
      TRUE ~ 0L
    ),
    adl_nonmissing = rowSums(!is.na(adl_binary)),
    adl_yes_count = rowSums(adl_binary, na.rm = TRUE),
    adl_any = case_when(
      adl_nonmissing == 0 ~ NA_integer_,
      adl_yes_count > 0 ~ 1L,
      TRUE ~ 0L
    )
  )

13. Figure 1: Medical aid coverage

build_age_group <- function(x) {
  case_when(
    x >= 15 & x <= 24 ~ "15-24",
    x >= 25 & x <= 34 ~ "25-34",
    x >= 35 & x <= 44 ~ "35-44",
    x >= 45 & x <= 54 ~ "45-54",
    x >= 55 & x <= 64 ~ "55-64",
    x >= 65 ~ "65+",
    TRUE ~ NA_character_
  )
}

med_summary <- adult_data_development |>
  mutate(
    age_group_plot = build_age_group(age_years),
    age_group_plot = factor(age_group_plot, levels = c("15-24", "25-34", "35-44", "45-54", "55-64", "65+")),
    medical_aid_yes = case_when(
      as.character(medical_aid) == "Yes" ~ 1,
      as.character(medical_aid) == "No" ~ 0,
      TRUE ~ NA_real_
    )
  ) |>
  filter(
    !is.na(age_group_plot),
    !is.na(population_group),
    !is.na(gender),
    !is.na(medical_aid_yes)
  ) |>
  group_by(age_group_plot, population_group, gender) |>
  summarise(
    medical_aid_coverage = mean(medical_aid_yes) * 100,
    n = n(),
    .groups = "drop"
  )

fig1_medical_aid <- med_summary |>
  ggplot(aes(x = age_group_plot, y = medical_aid_coverage, fill = population_group)) +
  geom_col(position = "dodge") +
  facet_wrap(~ gender, ncol = 2) +
  scale_y_continuous(labels = function(x) paste0(x, "%"), limits = c(0, 100)) +
  labs(
    title = "Medical Aid Coverage by Age, Population Group and Gender",
    x = "Age group",
    y = "Medical aid coverage (%)",
    fill = "Population group",
    caption = "Source: NIDS Wave 2 adult data. Author's calculations."
  ) +
  theme_minimal() +

    plot.title = element_text(face = "bold"),
    axis.text.x = element_text(angle = 45, hjust = 1),
